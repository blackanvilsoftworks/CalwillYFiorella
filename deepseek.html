Ejecución Dinámica de Funciones

Este ejemplo muestra cómo recorrer un array de objetos y ejecutar dinámicamente las funciones almacenadas en sus propiedades. Haz clic en el botón "Ejecutar Funciones" para ver el resultado.

Código JavaScript

const arrContainers = [
    {
        id: 'navbar-container',
        navbar: 'Menú Principal',
        title: '',
        icon: '',
        function: fillNavbar
    },
    {
        id: 'hero-section',
        navbar: 'Inicio',
        title: 'Bienvenido',
        icon: '',
        function: fillHeroSection
    },
];

// Función para ejecutar todas las funciones del array
function executeAllFunctions() {
    arrContainers.forEach(container => {
        // Verificamos que la propiedad function existe y es una función
        if (container.function && typeof container.function === 'function') {
            container.function(container); // Pasamos el objeto completo como parámetro
        }
    });
}

// Funciones de ejemplo
function fillNavbar(container) {
    const element = document.getElementById(container.id);
    if (element) {
        element.innerHTML = `
            <div class="logo">Mi Sitio</div>
            <nav>
                <ul>
                    <li><a href="#">${container.navbar}</a></li>
                    <li><a href="#">Servicios</a></li>
                    <li><a href="#">Contacto</a></li>
                </ul>
            </nav>
        `;
    }
}

function fillHeroSection(container) {
    const element = document.getElementById(container.id);
    if (element) {
        element.innerHTML = `
            <h2 class="hero-title">${container.title}</h2>
            <p class="hero-text">Esta es una sección hero creada dinámicamente mediante la ejecución de una función almacenada en un array de objetos.</p>
            <button class="btn">Saber más</button>
        `;
    }
}

// Ejecutar al hacer clic en el botón
document.getElementById('execute-btn').addEventListener('click', executeAllFunctions);
